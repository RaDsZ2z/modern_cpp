# 1.string(_view)及其字面量
```cpp
#include <iostream>
#include <string>
#include <string_view>

using namespace std::literals;                  // s
using namespace std::literals::string_literals; // sv
void f1(const char *c)
{
    std::cout << c << '\n';
}

void f2(std::string s)
{
    std::cout << s << '\n';
}
void f3(std::string_view s)
{
    std::cout << s << '\n';
}
int main()
{
    const char *c = "123";
    std::string s = "123"s;
    std::string_view sv = "123"sv;
    // 1.变量
    // f1
    f1(c);
    // f2
    f2(c);
    f2(s);
    // f3
    f3(c);
    f3(s);
    f3(sv);
    // 2.字面量
    // f1
    f1("123");
    // f2
    f2("123");
    f2("123"s);
    // f3
    f3("123");
    f3("123"s);
    f3("123"sv);
}
```
